using System.Linq;
using System.Web.Mvc;
using YourNamespace.Models;
public class HomeController : Controller
{
    private static readonly List<string> existingMobileNumbers = new List<string>
    {
        "1234567890",
        "0987654321"
    };
    [HttpPost]
    public JsonResult CheckMobileNumber(string mobileNumber)
    {
        bool isValid = !existingMobileNumbers.Contains(mobileNumber);
        string message = isValid ? "" : "Mobile number is already taken.";
        return Json(new { isValid = isValid, message = message });
    }
    [HttpPost]
    public ActionResult Submit(UserViewModel model)
    {

            return RedirectToAction("Success");

        return View(model);
    }

        return View();
